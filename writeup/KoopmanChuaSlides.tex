\begin{frame}{Motivation: Linear Tools for Nonlinear Motion}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.48\textwidth}
            \textbf{Linear systems}
            \begin{itemize}
                \item Eigenvalues: growth, decay, oscillation
                \item Eigenvectors: direction/shape
                \item Easy to work with
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Nonlinear systems}
            \begin{itemize}
                \item Linearize around fixed points - local information
                \item No global modal picture in state coordinates
                \item Structure may be hidden in nonlinear observables
            \end{itemize}
        \end{column}
    \end{columns}

    \vspace{0.45cm}
    \begin{block}{Koopman idea}
        Evolve observables of the state, not only the state itself.
    \end{block}
\end{frame}

\begin{frame}{The Koopman Operator}
    For a discrete-time system
    \[
        x_{n+1} = F(x_n), \qquad x_n \in \R^d,
    \]
    An observable is a function \(g:\R^d \to \C\):
    \[
        (\Koop g)(x) = g(F(x)).
    \]

    \vspace{0.25cm}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.48\textwidth}
            \begin{block}{Key property}
                \[
                    \Koop(a g_1 + b g_2)
                    = a\Koop g_1 + b\Koop g_2.
                \]
                Linear operator, nonlinear system.
            \end{block}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{block}{Price paid}
                The function space is generally infinite-dimensional.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{What Koopman Eigenfunctions Mean}
    A Koopman eigenfunction satisfies
    \[
        \Koop \xi_k = \lambda_k \xi_k,
        \qquad
        \xi_k(x_{n+1}) = \lambda_k \xi_k(x_n) \quad \text{(Scalar multiplication)}
    \]

    \vspace{0.15cm}
    Therefore,
    \[
        \xi_k(x_n) = \lambda_k^n \xi_k(x_0).
    \]

    \vspace{0.2cm}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.32\textwidth}
            \begin{block}{\(|\lambda| < 1\)}
                Decay
            \end{block}
        \end{column}
        \begin{column}{0.32\textwidth}
            \begin{block}{\(|\lambda| \approx 1\)}
                Sustained oscillation
            \end{block}
        \end{column}
        \begin{column}{0.32\textwidth}
            \begin{block}{\(\arg(\lambda)\)}
                Frequency per sample
            \end{block}
        \end{column}
    \end{columns}

    % \vspace{0.25cm}
    % \begin{alertblock}{Interpretation}
    %     Nonlinear motion in state space can become scalar multiplication in an eigenfunction coordinate.
    % \end{alertblock}
\end{frame}

\begin{frame}{A Small Exact Example}
    Consider
    \[
        \dot{x}_1 = \mu x_1,
        \qquad
        \dot{x}_2 = \lambda(x_2 - x_1^2).
    \]
    Add one nonlinear observable:
    \[
        \psi(x) =
        \begin{bmatrix}
            x_1 & x_2 & x_1^2
        \end{bmatrix}^{\top}
    \]
    The lifted state closes linearly:
    \[
        \frac{d}{dt}
        \begin{bmatrix}
            x_1 \\ x_2 \\ x_1^2
        \end{bmatrix}
        =
        \begin{bmatrix}
            \mu & 0 & 0 \\
            0 & \lambda & -\lambda \\
            0 & 0 & 2\mu
        \end{bmatrix}
        \begin{bmatrix}
            x_1 \\ x_2 \\ x_1^2
        \end{bmatrix}.
    \]

    \vspace{0.1cm}
    This is the finite-dimensional ideal EDMD tries to approximate.
\end{frame}

\begin{frame}{EDMD: Koopman from Snapshot Data}
    Choose a finite observable dictionary:
    \[
        \psi(x) =
        \begin{bmatrix}
            \psi_1(x) & \psi_2(x) & \cdots & \psi_{N_\psi}(x)
        \end{bmatrix}^{\top}.
    \]

    Build lifted snapshot matrices:
    \[
        \Psi_X =
        \begin{bmatrix}
            \psi(x_1) & \cdots & \psi(x_M)
        \end{bmatrix},
        \qquad
        \Psi_Y =
        \begin{bmatrix}
            \psi(x_2) & \cdots & \psi(x_{M+1})
        \end{bmatrix}.
    \]

    EDMD solves
    \[
        \Psi_Y \approx K \Psi_X.
    \]

    \vspace{0.15cm}
    \begin{block}{In this project}
        Least squares with truncated SVD; eigenvalues, modes, and eigenfunctions come from \(K\).
    \end{block}
\end{frame}

\begin{frame}{Chua's Circuit: Governing Equations}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.50\textwidth}
            State variables:
            \(x=V_{C_1},\,y=V_{C_2},\,z=I_L\)
            \[
                \dot x=\alpha[y-x-f(x)]
            \]
            \[
                RC_2\dot y=x-y+R_z
            \]
            \[
               \dot z =-\beta y
            \]

            \vspace{0.2cm}
            Chua diode nonlinearity:
            \[
                f(x) = G_b x
                  + \frac{G_a-G_b}{2}
                    \bigl(|x+E|-|x-E|\bigr)
            \]

            \vspace{0.15cm}
        \end{column}

    \end{columns}
    \begin{block}{Our implementation}
        Op-amp gyrator for the inductor; op-amp Chua diode; voltage shifting into the Arduino ADC range.
    \end{block}
\end{frame}
